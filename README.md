## Installation

```sh
npm install polestar-icons --save
```

### Usage

```jsx
import React from 'react';
import ReactDOM from 'react-dom';
import Icon from 'polestar-icons';

const App = () => (
    <Icon name="icon-name" />
);

ReactDOM.render(
  <App />,
  document.getElementById('app')
);
```
